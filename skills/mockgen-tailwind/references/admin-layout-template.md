# Admin Layout Architecture — Node.js + Alpine.js + HTMX

The mockup server uses **Express.js** to assemble pages from partials server-side,
**HTMX** for partial content swapping during navigation, and **Alpine.js** for
interactive UI components (dropdowns, dark mode toggle).

---

## File Structure

```
mockup/
  server.js                        # Express server
  package.json                     # Node.js dependencies
  MOCKUP.html                      # Landing/index page (static)
  partials/
    shell.html                     # Page shell (assembled by server)
    header.html                    # Top header (Alpine.js powered)
    footer.html                    # Bottom footer
    sidebar-{role_snake_case}.html # One sidebar per role
  {role_snake_case}/
    content/
      home.html                    # Content fragment (no layout)
      profile.html
      account.html
      notifications.html
      {module_snake_case}.html
      {module_snake_case}_detail.html
      {module_snake_case}_create.html
      {module_snake_case}_edit.html
      ...
```

## Placeholders Reference

| Placeholder | Description |
|-------------|-------------|
| `{{APP_NAME}}` | Application name (e.g., "Hub Middleware") |
| `{{PAGE_TITLE}}` | Screen title (e.g., "Location Information") |
| `{{ROLE}}` | Role in snake_case (e.g., "hub_administrator") |
| `{{SIDEBAR_MENU}}` | Navigation menu items HTML |
| `{{BREADCRUMB}}` | Breadcrumb trail HTML |
| `{{MAIN_CONTENT}}` | Screen-specific content |
| `{{COPYRIGHT_YEAR}}` | Current year |
| `{{VERSION}}` | Latest version from PRD.md |
| `{{DESIGN_PRIMARY}}` | Primary color hex from design system |
| `{{DESIGN_SECONDARY}}` | Secondary color hex from design system |
| `{{DESIGN_ACCENT}}` | Accent/CTA color hex |
| `{{DESIGN_BG}}` | Background color hex |
| `{{DESIGN_TEXT}}` | Text color hex |
| `{{HEADING_FONT}}` | Heading font |
| `{{BODY_FONT}}` | Body font |
| `{{GOOGLE_FONTS_URL}}` | Google Fonts import URL |

## server.js Template

```javascript
const express = require('express');
const fs = require('fs');
const path = require('path');

const app = express();
const PORT = process.env.PORT || 3000;
const BASE_DIR = __dirname;

// Serve static assets (images, etc.)
app.use('/static', express.static(path.join(BASE_DIR, 'static')));

// ── Helpers ──────────────────────────────────────────────────────────────────

function readFile(relPath) {
  const fullPath = path.join(BASE_DIR, relPath);
  if (!fs.existsSync(fullPath)) return null;
  return fs.readFileSync(fullPath, 'utf8');
}

function assemblePage(role, page) {
  const shell   = readFile('partials/shell.html');
  const header  = readFile('partials/header.html');
  const sidebar = readFile(`partials/sidebar-${role}.html`);
  const footer  = readFile('partials/footer.html');
  const content = readFile(`${role}/content/${page}.html`);

  if (!shell || !header || !sidebar || !footer) {
    return '<h1>Missing partials. Run mockup generator again.</h1>';
  }
  if (!content) {
    return shell
      .replace('{{HEADER}}', header)
      .replace('{{SIDEBAR}}', sidebar)
      .replace('{{CONTENT}}', '<div class="p-6 text-red-500">Page not found: ' + page + '</div>')
      .replace('{{FOOTER}}', footer);
  }
  return shell
    .replace('{{HEADER}}', header)
    .replace('{{SIDEBAR}}', sidebar)
    .replace('{{CONTENT}}', content)
    .replace('{{FOOTER}}', footer);
}

// ── Routes ───────────────────────────────────────────────────────────────────

// Landing page
app.get('/', (req, res) => {
  res.sendFile(path.join(BASE_DIR, 'MOCKUP.html'));
});

// Role home redirect
app.get('/:role', (req, res) => {
  res.redirect(`/${req.params.role}/home`);
});

// Full page (server-assembled from partials + content)
app.get('/:role/:page', (req, res) => {
  const { role, page } = req.params;
  const html = assemblePage(role, page);
  res.send(html);
});

// HTMX content-only endpoint (returns fragment for in-page swaps)
app.get('/api/content/:role/:page', (req, res) => {
  const { role, page } = req.params;
  const content = readFile(`${role}/content/${page}.html`);
  if (!content) return res.status(404).send('<div class="p-6 text-red-500">Content not found.</div>');
  res.send(content);
});

// ── Start ─────────────────────────────────────────────────────────────────────

app.listen(PORT, () => {
  console.log('');
  console.log('  Mockup server running at:');
  console.log(`  → http://localhost:${PORT}`);
  console.log('');
  console.log('  Open the URL above in your browser to review mockups.');
  console.log('');
});
```

## partials/shell.html Template

The shell is the outer HTML document. It is assembled server-side and NOT a partial
served to the browser — the server substitutes `{{HEADER}}`, `{{SIDEBAR}}`,
`{{CONTENT}}`, and `{{FOOTER}}` before responding.

Replace `{{ROLE}}` with the role's snake_case name. The `hx-get` attribute fetches only
the content fragment; `hx-push-url` updates the browser URL so the page is bookmarkable.

```html
<aside class="w-64 bg-white dark:bg-slate-800 border-r border-slate-200 dark:border-slate-700 fixed top-16 bottom-0 left-0 overflow-y-auto z-20">
  <nav class="py-4" x-data="{ active: window.location.pathname }">

    <!-- Home -->
    <a href="/{{ROLE}}/home"
       hx-get="/api/content/{{ROLE}}/home"
       hx-target="#content-area"
       hx-swap="innerHTML"
       hx-push-url="/{{ROLE}}/home"
       @click="active = '/{{ROLE}}/home'"
       :class="active.startsWith('/{{ROLE}}/home') ? 'bg-primary/5 text-primary border-r-2 border-primary font-medium' : 'text-slate-600 dark:text-slate-400 hover:bg-slate-50 dark:hover:bg-slate-700 hover:text-slate-900'"
       class="flex items-center gap-3 px-4 py-2.5 text-sm transition-colors duration-200 cursor-pointer">
      <svg class="w-5 h-5 shrink-0" fill="none" stroke="currentColor" viewBox="0 0 24 24">
        <path stroke-linecap="round" stroke-linejoin="round" stroke-width="2"
          d="M3 12l2-2m0 0l7-7 7 7M5 10v10a1 1 0 001 1h3m10-11l2 2m-2-2v10a1 1 0 01-1 1h-3m-6 0a1 1 0 001-1v-4a1 1 0 011-1h2a1 1 0 011 1v4a1 1 0 001 1m-6 0h6"/>
      </svg>
      <span>Home</span>
    </a>

    <!-- Module separator -->
    <div class="px-4 pt-4 pb-1">
      <span class="text-xs font-semibold text-slate-400 uppercase tracking-wider">Menu</span>
    </div>

    <!-- Module menu items: repeat for each module assigned to this role -->
    {{SIDEBAR_MENU}}

  </nav>
</aside>
```

## Content Fragment Template

Content files contain **only the page content area** — no `<html>`, `<head>`, `<body>`,
layout, or Tailwind config. They are loaded via HTMX and injected into `#content-area`.

```html

<!-- User story traceability: {{US_TAG}} [{{US_VERSION}}] -->
```

## Link Rules Inside Content Fragments

### Navigation links (within the app)

Use HTMX attributes for in-app navigation — never full page reloads for screen transitions:

```html
<a href="/{{ROLE}}/{{TARGET_PAGE}}"
   hx-get="/api/content/{{ROLE}}/{{TARGET_PAGE}}"
   hx-target="#content-area"
   hx-swap="innerHTML"
   hx-push-url="/{{ROLE}}/{{TARGET_PAGE}}"
   class="...">
  View Details
</a>
```

### Image links — MUST open in new tab

```html
<a href="/static/images/sample.jpg" target="_blank" rel="noopener noreferrer" class="...">
  View Image
</a>
```

### PDF / document links — MUST open in new tab

```html
<a href="/static/docs/sample.pdf" target="_blank" rel="noopener noreferrer" class="...">
  <svg ...><!-- pdf icon --></svg>
  View PDF
</a>
```

### Action buttons in tables

```html
<!-- View / Details → HTMX navigation -->
<a href="/{{ROLE}}/{{MODULE}}_detail"
   hx-get="/api/content/{{ROLE}}/{{MODULE}}_detail"
   hx-target="#content-area"
   hx-swap="innerHTML"
   hx-push-url="/{{ROLE}}/{{MODULE}}_detail"
   class="text-xs px-2 py-1 rounded border border-primary text-primary hover:bg-primary/5">
  View
</a>

<!-- Edit → HTMX navigation -->
<a href="/{{ROLE}}/{{MODULE}}_edit"
   hx-get="/api/content/{{ROLE}}/{{MODULE}}_edit"
   hx-target="#content-area"
   hx-swap="innerHTML"
   hx-push-url="/{{ROLE}}/{{MODULE}}_edit"
   class="text-xs px-2 py-1 rounded border border-slate-300 text-slate-600 hover:bg-slate-50">
  Edit
</a>

  @click="confirm = true"
  x-show="!confirm"
  class="text-xs px-2 py-1 rounded border border-red-300 text-red-600 hover:bg-red-50">
  Delete

  <span class="text-xs text-red-600">Sure?</span>
  <button class="text-xs px-2 py-1 bg-red-600 text-white rounded">Yes, Delete</button>
  <button @click="confirm = false" class="text-xs px-2 py-1 border border-slate-300 rounded">Cancel</button>
</div>

<!-- Add New → HTMX navigation -->
<a href="/{{ROLE}}/{{MODULE}}_create"
   hx-get="/api/content/{{ROLE}}/{{MODULE}}_create"
   hx-target="#content-area"
   hx-swap="innerHTML"
   hx-push-url="/{{ROLE}}/{{MODULE}}_create"
   class="px-4 py-2 rounded-lg text-sm text-white"

```

## server.js Role Variable Injection

When assembling pages server-side, the server must inject `{{ROLE}}` into header and
sidebar partials before sending. Update the `assemblePage` function:

```javascript
function assemblePage(role, page) {
  const shell   = readFile('partials/shell.html');
  const header  = readFile('partials/header.html').replace(/\{\{ROLE\}\}/g, role);
  const sidebar = readFile(`partials/sidebar-${role}.html`);
  const footer  = readFile('partials/footer.html');
  const content = readFile(`${role}/content/${page}.html`);
  // ... rest of assembly
}
```
